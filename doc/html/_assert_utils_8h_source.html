22inline std::vector<DirectX::XMFLOAT3> ReadBackVertexBuffer(ID3D12Device* device,
23 ID3D12GraphicsCommandList* commandList,
24 ID3D12CommandQueue* commandQueue,
25 ID3D12Resource* vertexBuffer)
26{
27 // Describe and create the readback buffer.
28 D3D12_RESOURCE_DESC vertexBufferDesc = vertexBuffer->GetDesc();
29
30 D3D12_HEAP_PROPERTIES heapProps = {};
31 heapProps.Type = D3D12_HEAP_TYPE_READBACK;
32
33 Microsoft::WRL::ComPtr<ID3D12Resource> readbackBuffer;
34 HRESULT hr = device->CreateCommittedResource(
35 &heapProps,
36 D3D12_HEAP_FLAG_NONE,
37 &vertexBufferDesc,
38 D3D12_RESOURCE_STATE_COPY_DEST,
39 nullptr,
40 IID_PPV_ARGS(&readbackBuffer)
41 );
42
43 if (FAILED(hr)) {
44 throw std::runtime_error("Failed to create readback buffer.");
45 }
46
47 // Copy the vertex buffer contents to the readback buffer.
48 commandList->CopyResource(readbackBuffer.Get(), vertexBuffer);
49
50 // Execute the command list and wait for it to finish.
51 commandList->Close();
52 ID3D12CommandList* ppCommandLists[] = { commandList };
53 commandQueue->ExecuteCommandLists(_countof(ppCommandLists), ppCommandLists);
54
55 // Create a fence and an event handle for synchronization.
56 Microsoft::WRL::ComPtr<ID3D12Fence> fence;
57 UINT64 fenceValue = 1;
58 hr = device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
59 if (FAILED(hr)) {
60 throw std::runtime_error("Failed to create fence.");
61 }
62
63 HANDLE fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
64 if (!fenceEvent) {
65 throw std::runtime_error("Failed to create fence event.");
66 }
67
68 // Signal and wait for the GPU to finish.
69 hr = commandQueue->Signal(fence.Get(), fenceValue);
70 if (FAILED(hr)) {
71 CloseHandle(fenceEvent);
72 throw std::runtime_error("Failed to signal command queue.");
73 }
74
75 if (fence->GetCompletedValue() < fenceValue) {
76 hr = fence->SetEventOnCompletion(fenceValue, fenceEvent);
77 if (FAILED(hr)) {
78 CloseHandle(fenceEvent);
79 throw std::runtime_error("Failed to set event on completion.");
80 }
81 WaitForSingleObject(fenceEvent, INFINITE);
82 }
83
84 CloseHandle(fenceEvent);
85
86 // Map the readback buffer to CPU memory.
87 void* pData = nullptr;
88 D3D12_RANGE readRange = { 0, 0 }; // We do not intend to write to this resource on the CPU
89 hr = readbackBuffer->Map(0, &readRange, &pData);
90
91 if (FAILED(hr)) {
92 throw std::runtime_error("Failed to map readback buffer.");
93 }
94
95 // Access the data.
96 size_t vertexCount = vertexBufferDesc.Width / sizeof(DirectX::XMFLOAT3);
97 std::vector<DirectX::XMFLOAT3> vertices(vertexCount);
98 memcpy(vertices.data(), pData, vertexBufferDesc.Width);
99
100 // Unmap the resource.
101 D3D12_RANGE writtenRange = { 0, 0 }; // Indicate that we haven't written to the buffer
102 readbackBuffer->Unmap(0, &writtenRange);
103
104 return vertices;
105}
